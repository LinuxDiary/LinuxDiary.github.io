# lsyncd

  • 系统环境：CentOS 7
  • 软件版本：lsyncd-2.2.2
  • Github地址：https://github.com/axkibe/lsyncd (opens new window)
  • 源服务器：192.168.10.52
  • 目标服务器：192.168.10.54

# 安装

在源服务器上安装 lsyncd ，配置后启动 lsyncd 服务

yum安装

yum install epel-release.noarch
yum install lsyncd

编译安装

安装依赖包，并创建 lsync 用户

yum -y install cmake gcc gcc-c++ autoconf automake lua lua-devel
useradd lsync
passwd lsync

切换到 lsync 普通用户，编译安装

su - lsync
mkdir -p ~/u01/lsyncd-2.2.2/{src,log,run,etc}
cd ~/u01/lsyncd-2.2.2/src
wget https://github.com/axkibe/lsyncd/archive/release-2.2.2.tar.gz
tar zxf release-2.2.2.tar.gz
cd lsyncd-release-2.2.2
cmake -DCMAKE_INSTALL_PREFIX=/home/lsync/u01/lsyncd-2.2.2
make && make install
cd ~/u01/
ln -s /home/lsync/u01/lsyncd-2.2.2/ /home/lsync/u01/lsyncd

# 配置文件

编辑配置文件，编译安装的配置文件位于 /home/lsync/u01/lsyncd/etc/lsyncd.conf ， yum 安装的配置文件位于 /etc/lsyncd.conf

本地同步

settings {
    logfile      ="/home/lsync/u01/lsyncd/log/lsyncd.log",
    statusFile   ="/home/lsync/u01/lsyncd/log/lsyncd.status",
    pidfile      ="/home/lsync/u01/lsyncd/run/lsyncd.pid",
    inotifyMode  = "CloseWrite",
    maxProcesses = 7,
    nodaemon = false,
    }

sync {
    default.rsync,
    source    = "/home/lsync/upload",
    target    = "/home/lsync/upload_bak",
    delay     = 15,
    -- excludeFrom = "/etc/rsyncd.d/rsync_exclude.lst",
    rsync     = {
        binary    = "/usr/bin/rsync",
        archive   = true,
        compress  = true,
        verbose   = true,
        bwlimit   = 10240,
        }
    }

通过ssh同步

settings {
    logfile      = "/home/lsync/u01/lsyncd/log/lsyncd.log",
    statusFile   = "/home/lsync/u01/lsyncd/log/lsyncd.status",
    pidfile      = "/home/lsync/u01/lsyncd/run/lsyncd.pid",
    inotifyMode  = "CloseWrite",
    maxProcesses = 1,
    nodaemon = false,
}

sync {
    default.rsyncssh,
    source    = "/home/lsync/master_backup",
    host      = "mysql@192.168.10.54",
    targetdir = "/home/mysql/master_backup",
    delay     = 15,
    -- excludeFrom = "/etc/rsyncd.d/rsync_exclude.lst",
    rsync     = {
        binary    = "/usr/bin/rsync",
        archive   = true,
        compress  = true,
        verbose   = true,
        bwlimit   = 10240,
        },
    ssh       = {
    port  =  62122
    }
}

# 启动

# 创建ssh秘钥

ssh 秘钥用于访问远程同步的目标服务器(192.168.10.54)，在本机(192.168.10.52)配置好后，向目标服务器导入 ssh key ，实现通过 ssh 秘钥免密远程同步

ssh-keygen -t rsa
ssh-copy-id -i ~/.ssh/id_rsa mysql@192.168.10.54

按提示输入远程服务器用户的密码

TIP

创建远程服务器秘钥后，需要 ssh 秘钥对应的远程服务器用户对 lsyncd 配置的远程服务器路径有读写权限。

# 启动服务

systemctl start lsyncd.service
systemctl enable lsyncd.service

如果是编译安装，则通过以下方式启动服务

su - lsync
/home/lsync/u01/lsyncd/bin/lsyncd /home/lsync/u01/lsyncd/etc/lsyncd.conf &
