# sendmail

# 安装并启动sendmail服务

yum -y install sendmail mailx
systemctl start sendmail
systemctl enable sendmail

# 配置发件人和服务器

这里使用公共的 163 邮件服务器发送邮件，而不使用服务器本身发送。编辑 /etc/mail.rc ，追加以下内容

set from=sendmail@163.com
set smtp=smtp.163.com
set smtp-auth-user=USER
set smtp-auth-password=PASSWORD
set smtp-auth=login

# 测试邮件发送

发送文本里的内容

mail -s 'Topic'  Receive@163.com  <  test.txt

直接发送文本信息

echo  'Content'  |  mail  -s  'Topic'  Receive@163.com
