# sysctl基本优化

# sysctl.conf

# 禁用ipv6
net.ipv6.conf.all.disable_ipv6 = 1
net.ipv6.conf.default.disable_ipv6 = 1

# 新建TCP连接时，发送SYN的重试次数，小于127，默认为6
net.ipv4.tcp_syn_retries = 1

# 被连接时，发送SYN+ACK的重试次数，小于255，默认为5
net.ipv4.tcp_synack_retries = 1

# keepalive 空闲时长，默认7200秒
net.ipv4.tcp_keepalive_time = 600

# 在 tcp_keepalive_time 之后没有收到对方应答时，keepalive 探测包的发送个数，超过3个就认为TCP连接已中断，默认为9
net.ipv4.tcp_keepalive_probes = 3

# keepalive 探测包发送的时间间隔，默认为75秒
net.ipv4.tcp_keepalive_intvl = 15

# 丢弃已激活的TCP连接前重试的次数，默认为15
net.ipv4.tcp_retries2 = 5

# FIN_WAIT_2的超时等待时间，默认为60秒
net.ipv4.tcp_fin_timeout = 30

# 系统允许的最大TIMEWAIT socket数，
net.ipv4.tcp_max_tw_buckets = 60000

# 允许重用TIMEWAIT socket为新连接，默认是2(仅用于loopback)，0为不允许
net.ipv4.tcp_tw_reuse = 1

# 当SYN队列溢出时，启用syncookie来处理，可防止SYN flood攻击，默认为1
net.ipv4.tcp_syncookies = 1

# 被连接端在回复ack之前，可接收最大的SYN队列，最小值128，可根据内存适当增加该值
net.ipv4.tcp_max_syn_backlog = 16384

# 为每个TCP分配的写缓冲区大小，min default max 默认为 4096 16384 64K~4MB(根据内存大小)
net.ipv4.tcp_wmem = 51200 131072 204800

# 为每个TCP分配的读缓冲区大小，min default max 默认为 4096 87380 87380~6MB(根据内存大小)
net.ipv4.tcp_rmem = 51200 131072 204800

# 定义本地的TCP/UDP端口可用范围，默认32768~60999
net.ipv4.ip_local_port_range = 1024 65000

# 系统级的每个端口的监听(listen)队列长度，也是系统全连接最大队列数，默认为128
net.core.somaxconn = 16384

# 每个网络接口接收数据包的速率比内核处理这些包的速率快时，允许送到队列的最大数目，一旦超过将被丢弃，默认为300
net.core.netdev_max_backlog = 16384

# 允许超过CommitLimit继续申请内存，因为申请的内存不一定会全部使用，开启该项可用最大限度的使用内存
vm.overcommit_memory = 1

# TCP连接超时时间，最后不低于tcp_keepalive_time，默认是432000秒(5天)
net.netfilter.nf_conntrack_tcp_timeout_established = 1200

其中 tcp_tw_recycle 参数已从新版内核中删除，且在 NAT 环境下配置该参数为 1 可能会出错 tcp_max_orphans 是为了防御简单 DDOS 攻击，一般不设置

net.nf_conntrack_max 和 net.netfilter.nf_conntrack_max 为防火墙连接跟踪表的最大值，正确设置方法为

vi /etc/modprobe.d/nf_conntrack.conf
options nf_conntrack hashsize=262144

CentOS7 下 nf_conntrack_max=hashsize*8 ，重启 iptables 后 nf_conntrack_max 会被重置为默认。

# sysctl命令

立即生效内核参数

sysctl -p /etc/sysctl.conf

查看所有内核参数

sysctl -a

临时修改内核参数

sysctl -w VARIABLE=VALUE
