# zabbix服务器

docker-compose-zabbix-server-5.0.yml

3.8版本需要使用v2版本的docker-compose

version: "3.8"
services:
  mysql-server:
    image: mysql:8.0
    container_name: mysql-server
    restart: always
    environment:
      MYSQL_DATABASE: zabbix
      MYSQL_USER: zabbix
      MYSQL_PASSWORD: Abc@123456
      MYSQL_ROOT_PASSWORD: Abc@123456
    command:
      - --character-set-server=utf8
      - --collation-server=utf8_bin
      - --default-authentication-plugin=mysql_native_password
    networks:
      - zabbix-net
    volumes:
      - "/root/u01/zabbix/mysql:/var/lib/mysql"
      - "/etc/localtime:/etc/localtime"
    ports:
      - "3306:3306"

  zabbix-java-gateway:
    image: zabbix/zabbix-java-gateway:alpine-5.0-latest
    container_name: zabbix-java-gateway
    volumes:
      - "/etc/localtime:/etc/localtime"
    ports:
      - "10052:10052"
    restart: always
    networks:
      - zabbix-net

  zabbix-server-mysql:
    image: zabbix/zabbix-server-mysql:alpine-5.0-latest
    container_name: zabbix-server-mysql
    networks:
      zabbix-net:
        ipv4_address: 172.18.0.10
    restart: always
    depends_on:
      - mysql-server
    environment:
      DB_SERVER_HOST: mysql-server
      MYSQL_DATABASE: zabbix
      MYSQL_USER: zabbix
      MYSQL_PASSWORD: Abc@123456
      MYSQL_ROOT_PASSWORD: Abc@123456
      ZBX_JAVAGATEWAY: zabbix-java-gateway
      ZBX_JAVAGATEWAY_ENABLE: "true"
    ports:
      - "10051:10051"
    volumes:
      - "/etc/localtime:/etc/localtime"
      - "/root/u01/zabbix/alertscripts:/usr/lib/zabbix/alertscripts"
      - "/root/u01/zabbix/externalscripts:/usr/lib/zabbix/externalscripts"
    links:
      - mysql-server:mysql
      - zabbix-java-gateway:zabbix-java-gateway

  zabbix-web-nginx-mysql:
    image: zabbix/zabbix-web-nginx-mysql:alpine-5.0-latest
    container_name: zabbix-web-nginx-mysql
    networks:
      - zabbix-net
    restart: always
    depends_on:
      - mysql-server
      - zabbix-server-mysql
    environment:
      DB_SERVER_HOST: mysql-server
      MYSQL_DATABASE: zabbix
      MYSQL_USER: zabbix
      MYSQL_PASSWORD: Abc@123456
      MYSQL_ROOT_PASSWORD: Abc@123456
      ZBX_SERVER_HOST: zabbix-server
      PHP_TZ: "Asia/Shanghai"
    ports:
      - "80:8080"
    volumes:
      - /etc/localtime:/etc/localtime
      - /root/u01/zabbix/fonts:/usr/share/zabbix/assets/fonts
    links:
      - mysql-server:mysql
      - zabbix-server-mysql:zabbix-server

  filerun:
    image: afian/filerun
    container_name: filerun
    networks:
      - zabbix-net
    restart: always
    environment:
      FR_DB_HOST: mysql-server
      FR_DB_PORT: 3306
      FR_DB_NAME: filerun
      FR_DB_USER: filerun_user
      FR_DB_PASS: Abc@123456
      APACHE_RUN_USER: www-data
      APACHE_RUN_USER_ID: 33
      APACHE_RUN_GROUP: www-data
      APACHE_RUN_GROUP_ID: 33
    depends_on:
      - mysql-server
    links:
      - mysql-server:mysql
    ports:
      - "81:80"
    volumes:
      - /etc/localtime:/etc/localtime
      - /root/u01/filerun/html:/var/www/html
      - /root/u01/filerun/user-files:/user-files
      
networks:
  zabbix-net:
    ipam:
      driver: default
      config:
        - subnet: "172.18.0.0/16"

解决字体问题：

拷贝一份微软雅黑字体，重命名为DejaVuSans.ttf，拷贝到/root/u01/zabbix/fonts目录下即可

agent安装见https://linuxdiary.github.io/Docker/docker_app.html#zabbix-agent (opens new window)

# Aria2+AriaNG

version: "3.8"

services:

  Aria2-Pro:
    container_name: aria2-pro
    image: p3terx/aria2-pro
    environment:
      - PUID=101
      - PGID=101
      - UMASK_SET=022
      - RPC_SECRET=P3TERX
      - RPC_PORT=6800
      - LISTEN_PORT=6888
      - DISK_CACHE=64M
      - IPV6_MODE=false
      - UPDATE_TRACKERS=true
      - CUSTOM_TRACKER_URL=
      - TZ=Asia/Shanghai
    volumes:
      - /root/u01/aria2-config:/config
      - /root/u01/kodbox/data/files/aria2-downloads:/downloads #指定下载目录到kodbox的本地存储中，配置目录用户和kodbox相同，实现与kodbox打通
# If you use host network mode, then no port mapping is required.
# This is the easiest way to use IPv6 networks.
#    network_mode: host
    network_mode: bridge
    ports:
      - 6800:6800
      - 6888:6888
      - 6888:6888/udp
    restart: unless-stopped
# Since Aria2 will continue to generate logs, limit the log size to 1M to prevent your hard disk from running out of space.
    logging:
      driver: json-file
      options:
        max-size: 1m

# AriaNg is just a static web page, usually you only need to deploy on a single host.
  AriaNg:
    container_name: ariang
    image: p3terx/ariang
    command: --port 83 --ipv6
#    network_mode: host
    network_mode: bridge
    ports:
      - 83:83
    restart: unless-stopped
    logging:
      driver: json-file
      options:
        max-size: 1m
