# 目录规划

TIP

执行命令 tree ~/u01/mysql-5.6.39 –L 2 可查看目录结构

# 准备

# 删除系统自带的my.cnf

CentOS 上可能会默认安装 mariadb-libs ，在 /etc/ 下有如果有一个 my.cnf 就删除

mv /etc/my.cnf /etc/my.cnf.bak

TIP

MySQL 默认会加载在编译时指定的配置文件目录中的 my.cnf ，如 -DSYSCONFDIR=/home/mysql/u01/mysql-5.6.39/etc 。如果没有指定，MySQL 默认按照顺序查找并加载 my.cnf ，如果在多个 my.cnf 文件中存在相同的参数， MySQL 以最后读到的参数为准。使用 mysql --help | grep -A 1 'Default options' 可查看 my.cnf 的加载顺序

# 依赖包和工具

yum -y install unzip bzip2 screen cmake gcc gcc-c++ autoconf ncurses ncurses-devel libaio libaio-devel libxml openssl openssl-devel readline readline-devel bison

# 用户

创建专用于运行和管理 MySQL 服务的用户 mysql

useradd mysql
echo "PASSWORD" | passwd --stdin mysql

编辑 /etc/security/limits.d/20-nproc.conf 文件设置 mysql 用户的最大打开文件数和最大进程数

mysql   soft    nproc   65535
mysql   hard    nproc   65535
mysql   soft    nofile  65535
mysql   hard    nofile  65535

ulimit 值根据系统情况设置，可参考 cat /proc/sys/fs/file-max ，不能大于 file-max ，设置后最好重启一次系统

# 编译

切换到 mysql 用户

su - mysql

创建所需要的目录和文件

mkdir -p ~/u01/mysql-5.6.39/{src,data,ibdata,iblog,binlog,etc,log,run,tmp}
touch ~/u01/mysql-5.6.39/log/{error.log,slow.log}

TIP

MySQL 所要用到的目录和文件需要事先创建好， MySQL 在启动时不会主动去创建这些文件，例如 err.log 和 tmp 目录，否则可能会报错，提示文件或目录不存在

编译后安装

cd ~/u01/mysql-5.6.39/src
wget https://dev.mysql.com/get/Downloads/MySQL-5.6/mysql-5.6.39.tar.gz
tar zxf mysql-5.6.39.tar.gz
cd mysql-5.6.39
cmake . \
-DCMAKE_INSTALL_PREFIX=/home/mysql/u01/mysql-5.6.39 \
-DSYSCONFDIR=/home/mysql/u01/mysql-5.6.39/etc \
-DMYSQL_DATADIR=/home/mysql/u01/mysql-5.6.39/data \
-DMYSQL_UNIX_ADDR=/home/mysql/u01/mysql-5.6.39/run/mysql.sock \
-DWITH_INNOBASE_STORAGE_ENGINE=1 \
-DWITH_MYISAM_STORAGE_ENGINE=1 \
-DWITH_ARCHIVE_STORAGE_ENGINE=1 \
-DWITH_PARTITION_STORAGE_ENGINE=1 \
-DWITH_FEDERATED_STORAGE_ENGINE=1 \
-DWITH_BLACKHOLE_STORAGE_ENGINE=1 \
-DWITH_ZLIB=bundled \
-DEXTRA_CHARSETS=all \
-DDEFAULT_CHARSET=utf8 \
-DDEFAULT_COLLATION=utf8_general_ci \
-DENABLED_LOCAL_INFILE=1 \
-DWITH_READLINE=1 \
-DWITH_EMBEDDED_SERVER=1 \
-DWITH_READLINE=1 \
-DWITH_SSL=yes

make -j4 && make install

TIP

-j4 参数表示启用 4 个线程来执行编译操作，可有效减少编译时间

编译参数 说明
-DCMAKE_INSTALL_PREFIX MySQL安装目录
-DSYSCONFDIR my.cnf所在的目录
-DMYSQL_DATADIR 数据目录
-DMYSQL_UNIX_ADDR socket文件路径
-DWITH_INNOBASE_STORAGE_ENGINE 是否启用innodb存储引擎，下同
-DWITH_MYISAM_STORAGE_ENGINE
-DWITH_ARCHIVE_STORAGE_ENGINE
-DWITH_PARTITION_STORAGE_ENGINE
-DWITH_FEDERATED_STORAGE_ENGINE
-DWITH_BLACKHOLE_STORAGE_ENGINE
-DWITH_ZLIB 指定zlib，bundled表示使用MySQL源码包带的zlib
-DEXTRA_CHARSETS 除了默认字符编码意外，还支持的字符编码
-DDEFAULT_CHARSET 默认字符编码
-DDEFAULT_COLLATION 默认字符校验集
-DENABLED_LOCAL_INFILE 是否允许导入本地数据文件
-DWITH_READLINE 是否启用readline
-DWITH_EMBEDDED_SERVER 是否启用embedded server库
-DWITH_READLINE 是否启用readline
-DWITH_SSL=yes 是否启用SSL

TIP

也可以直接使用编译好的二进制包，例如 mysql-5.6.39-linux-glibc2.12-x86_64.tar.gz ，该二进制包需要 glibc2.12 版本支持，可使用 ldd --version 命令查看 glibc 版本

# 优化

创建根目录软连接

ln -s /home/mysql/u01/mysql-5.6.39 /home/mysql/u01/mysql

链接头文件和库文件

sudo ln -sf /home/mysql/u01/mysql/include /usr/include/mysql
sudo bash -c 'echo "/home/mysql/u01/mysql/lib" >> /etc/ld.so.conf.d/mysql.conf'
sudo ldconfig

设置环境变量

sudo bash -c 'echo "export PATH=/home/mysql/u01/mysql/bin:$PATH" >> /etc/profile.d/mysql.sh'
sudo chmod +x /etc/profile.d/mysql.sh
. /etc/profile.d/mysql.sh

TIP

MySQL 的环境变量路径放在 $PATH 前面，好处是就算系统存在其他的 mysql 命令，也会首先读取 /home/mysql/u01/mysql/bin 下的命令，而忽略其他的 mysql 命令

# 编写脚本

#!/bin/bash

. /etc/init.d/functions

mysql_user=root
port=3306
mysql_bin_path="/home/mysql/u01/mysql/bin"
mysql_socket="/home/mysql/u01/mysql/run/mysql.sock"
mysql_conf="/home/mysql/u01/mysql/etc/my.cnf"

start_mysql()
{
    if [ -e $mysql_socket ]; then
        echo "MySQL Server is running..."
        exit
    else
        $mysql_bin_path/mysqld_safe --defaults-file=$mysql_conf > /dev/null &
        if [[ $? = 0 ]];then
            sleep 2
            action "Start MySQL" /bin/true
        else
            action "Start MySQL" /bin/false
        fi
    fi
}

stop_mysql()
{
    if [ -e $mysql_socket ]; then
        $mysql_bin_path/mysqladmin -u$mysql_user -p --socket=$mysql_socket shutdown > /dev/null
        if [[ $? = 0 ]];then
            #sleep 2
            action "Stop MySQL" /bin/true
        else
            action "Stop MySQL" /bin/false
        fi
    else
        echo "MySQL Server is stopped..."
        exit
    fi
}

restart_mysql()
{
    stop_mysql
    start_mysql
}

case $1 in
    start   )
        start_mysql
        ;;
    stop    )
        stop_mysql
        ;;
    restart )
        restart_mysql
        ;;
    *   )
        echo -e "Usage: $mysql_bin_path/mysqld_job {start|stop|restart}\n"
        ;;
esac

exit 0

# 设置或修改root密码

设置或修改时需要指定 socket 路径

mysqladmin -uroot password 'NEWPASS' --socket=/home/mysql/u01/mysql-multi/3306/run/mysql.sock

# 停止服务

同样需要指定 socket 路径

mysqladmin -u root -p PASSWORD --socket=/home/mysql/u01/mysql-multi/3306/run/mysql.sock shutdown

在服务器上登录 MySQL 多实例时，需要指定 socket 路径，同时最好也指定对应的配置文件，如果是远程登录，则是指定服务端口

mysql --socket=/home/mysql/u01/mysql-multi/3306/run/mysql.sock -uroot -p
