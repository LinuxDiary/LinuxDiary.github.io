# 邮件发送服务

安装邮件发送服务，在触发监控条件时，向指定邮箱发送邮件

sudo yum -y install sendmail mailx
sudo systemctl start sendmail
sudo systemctl enable sendmail

设置发件人和邮件服务器，编辑 /etc/mail.rc 追加以下内容

set from=sendmail@163.com
set smtp=smtp.163.com
set smtp-auth-user=USER
set smtp-auth-password=PASSWORD
set smtp-auth=login

# 运行状态监控

#!/bin/bash

export PATH=/home/mysql/u01/mysql/bin:/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin

user='root'
password='PASSWORD'
host=
port=
socket='/home/mysql/u01/mysql-multi/3306/run/mysql.sock'
tmp='/home/mysql/tmp'
tmpfile='ruv6GKje2q'
email='sendmail@163.com'

if [[ ! -d $tmp ]]; then
    /usr/bin/mkdir -p $tmp
fi

mysql_check=$(mysqladmin -u$user -p$password --socket=$socket ping 2> /dev/null)

if [[ $mysql_check != "mysqld is alive" ]]; then
    echo "
================================================================================
= MySQL is not alive, please check the server !!!
================================================================================
`date`
" > $tmp/$tmpfile
    cat $tmp/$tmpfile
    # mail -s 'MySQL is not alive' $email < $tmp/$tmpfile
fi

保存脚本并授权，脚本包含明文密码，赋予 700 权限

chmod 700 /home/mysql/mysql.chk.sh

添加 cron 任务，每 5 分钟执行一次

$ crontab -l
*/5 * * * * /bin/bash /home/mysql/mysql.chk.sh

# Sleep数监控

#!/bin/bash

export PATH=/home/mysql/u01/mysql/bin:/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin

user='root'
password='PASSWORD'
host=
port=
socket='/home/mysql/u01/mysql-multi/3306/run/mysql.sock'
tmp='/home/mysql/tmp'
tmpfile='ruv6GKje2q'
email='sendmail@163.com'

if [[ ! -d $tmp ]]; then
    /usr/bin/mkdir -p $tmp
fi

mysql_check=$(mysqladmin -u$user -p$password --socket=$socket ping 2> /dev/null)
mysql_command=$(mysql -u$user -p$password --socket=$socket -e 'show full processlist;')
sleep_count=$(echo "$mysql_command" | grep Sleep | wc -l)

if [[ $mysql_check != "mysqld is alive" ]]; then
    echo "
================================================================================
= MySQL is not alive, please check the server !!!
================================================================================
`date`
" > $tmp/$tmpfile
    cat $tmp/$tmpfile
    # mail -s 'MySQL is not alive' $email < $tmp/$tmpfile
    exit 1
fi

if [[ $sleep_count -gt 22 ]]; then
    echo "
================================================================================
= There is more than 22 Sleeps !!!
================================================================================
`date`
" > $tmp/$tmpfile
    echo "$mysql_command" >> $tmp/$tmpfile
    cat $tmp/$tmpfile
    # mail -s 'Too many Sleeps' $email < $tmp/$tmpfile
fi

授权然后添加计划任务

$ chmod 700 /home/mysql/sleep.chk.sh
$ crontab -l
*/5 * * * * /bin/bash /home/mysql/sleep.chk.sh

# Salve状态监控

#!/bin/bash

export PATH=/home/mysql/u01/mysql/bin:/usr/local/sbin:/usr/local/bin:/usr/sbin:/usr/bin

user='root'
password='PASSWORD'
host=
port=
socket='/home/mysql/u01/mysql-multi/3306/run/mysql.sock'
tmp='/home/mysql/tmp'
tmpfile='ruv6GKje2q'
email='sendmail@163.com'

if [[ ! -d $tmp ]]; then
    /usr/bin/mkdir -p $tmp
fi

mysql_check=$(mysqladmin -u$user -p$password --socket=$socket ping 2> /dev/null)
slave_check=$(mysql -u$user -p$password --socket=$socket -e 'show slave status\G' 2> /dev/null)

io_status=$(echo "$slave_check" | grep "Slave_IO_Running:" | awk '{ print $2 }')
sql_status=$(echo "$slave_check" | grep "Slave_SQL_Running:" | awk '{ print $2 }')
seconds_behind_master=$(echo "$slave_check" | grep "Seconds_Behind_Master" | awk '{ print $2 }')

if [[ $mysql_check != "mysqld is alive" ]]; then
    echo "
================================================================================
= MySQL is not alive, please check the server !!!
================================================================================
`date`
" > $tmp/$tmpfile
    cat $tmp/$tmpfile
    # mail -s 'MySQL is not alive' $email < $tmp/$tmpfile
    exit 1
fi

if [[ $io_status != "Yes" || $sql_status != "Yes" ]]; then
    echo "
================================================================================
= Slave is down, please check the slave server !!!
================================================================================
`date`
" > $tmp/$tmpfile
    echo "$slave_check" >> $tmp/$tmpfile
    cat $tmp/$tmpfile
    # mail -s 'Slave is down' $email < $tmp/$tmpfile
elif [[ $seconds_behind_master -gt 60 ]]; then
    echo "
================================================================================
= Slave is working, but the slave is $seconds_behind_master seconds behind the master !!!
================================================================================
`date`
" > $tmp/$tmpfile
    echo "$slave_check" >> $tmp/$tmpfile
    cat $tmp/$tmpfile
    # mail -s 'Slave is behind master' $email < $tmp/$tmpfile
fi

授权然后添加计划任务

$ chmod 700 /home/mysql/slave.chk.sh
$ crontab -l
*/5 * * * * /bin/bash /home/mysql/slave.chk.sh

# 批量修改表引擎

本质上是使用 alter 命令对表执行变更存储引擎操作，首先生成待执行的 change_list.sql ，然后在 MySQL 中运行它

# 生成批量执行脚本

mysql --user=root --password -e "show table status from linuxd where Engine <> 'InnoDB\G"|grep Name|awk '{print "alter table "$2" engine=innodb;";}' > change_list.sql

change_list.sql 内容

alter table comments_comment engine=innodb;
alter table blog_tag engine=innodb;
···
alter table blog_post_tags engine=innodb;

# 执行转换脚本

执行前备份数据库。执行时间根据数据库大小决定，执行期间，最好停库。首先登陆 MySQL

screen -S alter
mysql -uroot -p

执行 change_list.sql 脚本

use linuxd;
source change_list.sql;

TIP

也可以使用 mysqldump 工具导出到 SQL 脚本文件，将 ENGINE=MyISAM 替换成 ENGINE=InnoDB ，然后再将数据导入到数据库中实现存储引擎变更，在 mysqldump 导出数据时需要全局锁表，如果可以最好停库操作。如果数据库体量大，这个过程可能需要较长时间。
