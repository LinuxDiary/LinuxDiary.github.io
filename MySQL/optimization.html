# 运行环境优化

# CPU

关闭 NUMA ， NUMA 是非一致性内存访问架构，不适合数据库服务的运行，可以在系统内核中关闭

CentOS6

编辑 /etc/grub.conf 文件，在 kernel 参数后面追加 numa=off

kernel /vmlinuz-2.6.32-431.el6.x86_64 ro root=UUID=2cccbdae-d43f-4109-9679-e4ab6c32afde rd_NO_LUKS rd_NO_LVM LANG=en_US.UTF-8 rd_NO_MD SYSFONT=latarcyrheb-sun16 crashkernel=auto  KEYBOARDTYPE=pc KEYTABLE=us rd_NO_DM rhgb quiet numa=off

CentOS7

编辑内核参数 /etc/default/grub 追加 numa=off

GRUB_CMDLINE_LINUX="crashkernel=auto rd.lvm.lv=centos/root rd.lvm.lv=centos/swap rhgb quiet numa=off"

重新生成 grub 配置，然后重启系统

grub2-mkconfig -o /boot/grub2/grub.cfg

# 内存

设置 vm.swappiness 参数，尽可能的不使用交换空间

echo "vm.swappiness = 1" > /etc/sysctl.conf
sysctl -p

# I/O

针对机械硬盘，更改 I/O 调度算法为 Deadline ；针对 SSD ，更改 I/O 调度算法为 NOOP

临时生效

echo deadline > /sys/block/sda/queue/scheduler

永久生效

grubby --update-kernel=ALL --args="elevator=deadline"

# 文件系统

文件系统是 xfs 或 ext3/4 ，使用挂载参数 noatime 和 nodiratime ，能一定程度上提供磁盘的性能

/dev/sda /mountlocation xfs default, noatime, nodiratime, data=writeback 0 0

# 操作系统内核

以 CentOS 7 为例，在 sysctl.conf 中添加以下参数

net.core.somaxconn = 65535
net.core.netdev_max_backlog = 20000
net.ipv4.tcp_max_syn_backlog = 65535

net.ipv4.tcp_fin_timeout = 30
net.ipv4.tcp_tw_recycle = 0
net.ipv4.tcp_tw_reuse = 1

net.core.wmem_default = 212992
net.core.wmem_max = 16777216
net.core.rmem_default = 212992
net.core.rmem_max = 16777216

net.ipv4.tcp_keepalive_time = 30
net.ipv4.tcp_keepalive_intvl = 30
net.ipv4.tcp_keepalive_probes = 3

# 最大共享内存段大小，默认是物理内存的两倍，可以设置成略小于物理内存
kernel.shmmax = 68719476736

vm.swappiness = 1
