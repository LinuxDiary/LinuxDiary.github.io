# SQL

# 分类

  • DQL(数据查询语言)：SELECT FROM
  • DML(数据操作语言)：UPDATE、DELETE、INSERT INTO
  • DDL(数据定义语言)：CREATE [DATABASE,TABLE,INDEX]、ALTER [DATABASE,TABLE]、DROP [DATABASE,TABLE,INDEX]
  • DCL(数据控制语言)：GRANT、REVOKE、COMMIT、ROLLBACK

# 库相关

查看MySQL实例中所有数据库

SHOW DATABASES;

使用通配符

SHOW DATABASES like '%in%';

创建数据库

CREATE DATABASE linuxd DEFAULT CHARACTER SET utf8 COLLATE utf8_general_ci;

如果使用MySQL实例默认的字符集，可以直接建库

CREATE DATABASE linuxd;

查看数据库的声明信息

SHOW CREATE DATABASE linuxd;

查看当前use的数据库

SELECT database();

删除数据库

DROP DATABASE linuxd;

# 表相关

创建表

建表时可以指定表的存储引擎(ENGINE)，字符集(DEFAULT CHARSET)，行格式(ROW_FORMAT)等，不指定则使用系统默认值

CREATE TABLE linuxd.test (
    id int(4) NOT NULL AUTO_INCREMENT COMMENT 'master',
    name char(20) NOT NULL,
    age int(4) NOT NULL,
    PRIMARY KEY (id)
    );

查看建表的声明信息

SHOW CREATE TABLE linuxd.test;

查看库中的表清单

SHOW TABLES FROM linuxd;

查看表结构

DESC linuxd.test;
或
SHOW COLUMNS FROM linuxd.test;

如果要查看完整的表结构信息

SHOW FULL COLUMNS FROM linuxd.test;

创建主键

如果在建表时没有创建主键，则可以添加主键并指定自增

ALTER TABLE linuxd.test ADD PRIMARY KEY (id),CHANGE id id int(4) NOT NUll AUTO_INCREMENT;
或
mysql>ALTER TABLE linuxd.test CHANGE id id int(4) NOT NUll PRIMARY KEY AUTO_INCREMENT;

删除主键

删除主键时如果是带有 AUTO_INCREMENT ，那么删除会报错。如上所述，删除主键的同时 CHANGE 字段的属性即可。

ALTER TABLE linuxd.test DROP PRIMARY KEY;
ERROR 1075 (42000): Incorrect table definition; there can be only one auto column and it must be defined as a key

正确删除方法

ALTER TABLE linuxd.test DROP PRIMARY KEY, CHANGE id id int(4) NOT NULL;

创建联合主键

创建联合主键只需要在建表时在 PRIMARY KEY 中指定多个字段即可，如： PRIMARY KEY (Host,User)

DESC mysql.user;
+------------------------+-----------------------------------+------+-----+---------+-------+

| Host                   | char(60)                          | NO   | PRI |         |       |
| User                   | char(16)                          | NO   | PRI |         |       |
| Password               | char(41)                          | NO   |     |         |       |

创建普通索引

ALTER TABLE linuxd.test ADD INDEX index_age(age);
或
CREATE INDEX index_age ON linuxd.test(age);

创建唯一索引

ALTER TABLE linuxd.test ADD UNIQUE INDEX index_age(age);
或
CREATE UNIQUE INDEX index_age ON linuxd.test(age);

删除索引

普通索引和唯一索引方法一样

DROP INDEX index_age on linuxd.test;
或
ALTER TABLE linuxd.test DROP INDEX index_age;

修改索引

先删除，再创建

ALTER TABLE linuxd.test DROP INDEX index_age, CHANGE age age int(4) NOT NULL;
CREATE INDEX index_age ON linuxd.test(age);

使用索引前缀

如果表中的列比较大，前n个字符已经唯一，可以使用该列的前 n 个字符(索引前缀 index prefix )作为索引，提高查询效率。在 InnoDB 存储引擎中， innodb_large_prefix 参数可以允许使用更大的索引前缀( 3072 字节)，默认是 767 字节。

CREATE INDEX index_name ON linuxd.test(name(6));

查看索引

其中 PRI 表示主键， MUL 表示普通索引， UNI 表示唯一索引

DESC linuxd.test;

| Field | Type     | Null | Key | Default | Extra          |
+-------+----------+------+-----+---------+----------------+
| id    | int(4)   | NO   | PRI | NULL    | AUTO_INCREMENT |
| name  | char(20) | NO   | MUL | NULL    |                |
| age   | int(4)   | NO   | UNI | NULL    |                |
+-------+----------+------+-----+---------+----------------+

创建联合索引

与联合主键一样，指定多个字段即可

CREATE INDEX index_name_age ON linuxd.test(name,age);

索引往往创建在查询条件 WHERE 语句所跟的条件列中,且只适用于 and 组合， or 组合无效。联合索引遵循最左原则，即可以满足条件1,、条件1条件2、条件1条件2条件3，但是满足条件2条件3则不能正确匹配。

更改表名

RENAME TABLE oldtable TO newtable;
或
ALTER TABLE oldtable RENAME TO newtable;

更改字段

ADD 增加字段，并指明属性 NOT NULL、PRIMARY KEY 等. AFTER 表示在 name 后面增加字段， FIRST 表示把字段加在第一个

ALTER TABLE linuxd.test ADD sex char(4) NOT NULL AFTER name;

MODIFY 修改字段属性，只能修改字段属性，同样可以使用 AFTER|FIRST 移动位置到 age 字段后面

ALTER TABLE linuxd.test MODIFY sex char(20) NOT NULL AFTER age;

CHANGE 修改字段名和属性，并放在第一个字段的位置

ALTER TABLE linuxd.test CHANGE sex sex_new char(10) NOT NULL FIRST;

DROP 删除字段，也就是删除列，可以一次删除多个

ALTER TABLE linuxd.test DROP COLUMN name,DROP COLUMN age;

# 数据相关

插入数据

可以省略 (id,name,age) ，但前提是 VALUES 后面的排列要与表的字段排列顺序一致

NSERT INTO linuxd.test(id,name,age) VALUES (1,'jack',28);

如果 id 是自增，那么可以不指定 id 字段和值，一次插入多行数据的效率优于多行数据一次插入一行。

INSERT INTO linuxd.test(name,age) VALUES ('jack',28),('john',29),('lisa',30),('stephen',31);

普通查询

SELECT id,name FROM linuxd.test

条件查询

不输出查询结果的前 2 行，从第 3 行开始输出 3 行数据， ORDER BY 以某列排序， ASC 正序排列， DESC 反序排列。 LIMIT 查询前n行，语法是 LIMIT [offset,] rows | rows OFFSET offset ， rows 指定了返回记录的最大数目 offset 偏移量默认为 0

SELECT id,name FROM linuxd.test ORDER BY id ASC LIMIT 2,3;

等同于上面语句

SELECT id,name FROM linuxd.test ORDER BY id ASC LIMIT 3 OFFSET 2;

多表联查

多表联查要求查询的表有相同的列，且这个列的值必须唯一。

USE linuxd
SELECT test.name,test1.age,test2.sex FROM test,test1,test2 WHERE test1.id=test2.id and test.age=test2.age ORDER BY id ASC;

修改数据

UPDATE linuxd.test SET name='john' WHERE id=2;

删除数据

DELETE FROM linuxd.test WHERE id=1;

清空数据

两条语句都能够清空表，区别在于 DELETE 是逻辑的一条一条的从表中删除数据，如果表的体量大，速度会很慢； TRUNCATE 相当于直接清除数据表文件，速度更快。

DELETE FROM linuxd.test;
或
TRUNCATE TABLE linuxd.test;

防止误删误改

在 UPDATE 或 DELETE 数据的时候，如果不用 WHERE 指定条件，操作的将是整个列的数据。可以在 mysql 命令登陆时加 -U 参数强制加条件，防止人为误操作。可以在 Linux 中把 mysql 命令作成别名

mysql -h192.168.2.22 -P3306 -uroot -p -U

效果如下

DELETE FROM linuxd.test;
ERROR 1175 (HY000): You are using safe update mode and you tried to update a table without a WHERE that uses a KEY column

# 用户相关

查看当前客户端登陆用户

SELECT user();
+-------------------+
| user()            |
+-------------------+
| root@192.168.2.20 |
+-------------------+

查看当前登陆对应服务端的用户

SELECT current_user();
+------------------+
| current_user()   |
+------------------+
| root@192.168.2.% |
+------------------+

创建用户并授权

MySQL 中对用户授权是对 user@host 的组合进行授权， host 的形式可以是 IP 、域名、网段(192.168.1.%或192.168.1.0/255.255.255.0)、 %(代表所有)

如果是管理员账号，可以授权 ALL ；如果是应用账号，一般只授权 SELECT,INSERT,UPDATE,DELETE 即可；如果是主从复制账号，就授权 REPLICATION SLAVE 。

CREATE USER 'stj'@'%' IDENTIFIED BY 'password';
GRANT ALL ON *.* TO 'stj'@'%';

创建用户的同时进行赋权

GRANT ALL ON *.* TO 'stj'@'%' IDENTIFIED BY 'password';
FLUSH PRIVILEGES;

创建赋权用户

只需要在授权时加上 WITH GRANT OPTION 即可。

GRANT ALL ON *.* TO 'stj'@'%' IDENTIFIED BY 'password' WITH GRANT OPTION;
FLUSH PRIVILEGES;

GRANT OPTION 表示对用户赋予赋权权限，可以对其他用户进行 GRANT 赋权。并且只能对全库 *.* 使用这个参数才能生效，如果是对某一个数据库表使用了这个选项，则不会生效。检查生效主要看两个表的字段( Grant_priv 和 IS_GRANTABLE )的值

SELECT user,host,Grant_priv FROM mysql.user;
+------+-------------+------------+
| user | host        | Grant_priv |
+------+-------------+------------+
| root | localhost   | Y          |
| root | multi-mysql | Y          |
| root | 127.0.0.1   | Y          |
| root | ::1         | Y          |
| root | 192.168.2.% | Y          |
| stj  | %           | N          |
+------+-------------+------------+

SELECT * FROM information_schema.user_privileges WHERE GRANTEE="'stj'@'%'";
+-----------+---------------+----------------+--------------+
| GRANTEE   | TABLE_CATALOG | PRIVILEGE_TYPE | IS_GRANTABLE |
+-----------+---------------+----------------+--------------+
| 'stj'@'%' | def           | USAGE          | NO           |
+-----------+---------------+----------------+--------------+

如果 Grant_priv 和 IS_GRANTABLE 对应用户的值都是 Y 或 YES ，代表该用户确定有赋权权限， user_privileges 表中的字段 PRIVILEGE_TYPE 表示该用户可以赋予其他用户何种类型的权限，而 IS_GRANTABLE 字段表示该用户是否有赋予其他用户 PRIVILEGE_TYPE 的权限

回收权限

回收 stj@localhost 的所有权限

REVOKE ALL FROM 'stj'@'localhost';

回收 stj@localhost 的所有权限和赋权权限

REVOKE ALL,GRANT OPTION FROM 'stj'@'localhost';
FLUSH PRIVILEGES;

重命名用户

RENAME USER olduser to newuser

删除用户

DROP USER "root"@"localhost";
FLUSH PRIVILEGES;

如果只是执行了下面语句，并不能真正删除用户

DELETE from mysql.user where user='root' and user='TEST';

这条语句只是删除了 user 表中的一条记录，实际上跟用户相关的数据在 information_schema.user_privileges 表中还有，应该使用 DROP 才能完整删除用户。
