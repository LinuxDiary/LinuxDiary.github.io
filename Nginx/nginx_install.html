# 安装

# 依赖

sudo yum install -y prce pcre-devel openssl openssl-devel policycoreutils-python gcc gcc-c++ make

# 运行用户

创建一个普通用户来运行 Nginx ，而不使用 root

useradd nginx
passwd nginx

设置 nginx 用户的最大打开文件数和最大进程数，设置完成后最好重启一遍系统

sudo vi /etc/security/limits.d/20-nproc.conf
nginx   soft    nproc   65535
nginx   hard    nproc   65535
nginx   soft    nofile  65535
nginx   hard    nofile  65535

# 编译

mkdir -p /home/nginx/u01/nginx-1.13.12/src
cd /home/nginx/u01/nginx-1.13.12/src
wget http://nginx.org/download/nginx-1.13.12.tar.gz
tar xf nginx-1.13.12.tar.gz
cd nginx-1.13.12
./configure \
--prefix=/home/nginx/u01/nginx-1.13.12 \
--user=nginx \
--group=nginx \
--pid-path=/home/nginx/u01/nginx-1.13.12/run/nginx.pid \
--lock-path=/home/nginx/u01/nginx-1.13.12/run/nginx.lock \
--error-log-path=/home/nginx/u01/nginx-1.13.12/logs/error.log \
--http-log-path=/home/nginx/u01/nginx-1.13.12/logs/access.log \
--with-http_ssl_module \
--with-http_stub_status_module \
--with-http_gzip_static_module \
--with-http_stub_status_module \
--with-pcre \
--with-file-aio \
--with-http_realip_module \
--with-http_v2_module \
--with-stream

make && make install

# 软链接

为 Nginx 创建一条软链接，方便管理

ln -s /home/nginx/u01/nginx-1.13.12 /home/nginx/u01/nginx
sudo ln -s /home/nginx/u01/nginx/sbin/nginx /usr/bin/nginx

# 服务端口

非 root 用户不能使用 1024 以下的端口，有2种解决办法

# sudo

sudo `which nginx`

# iptables端口转发

重定向 80 端口到 8888 端口

sudo iptables -t nat -A PREROUTING -i eth0 -p tcp --dport 80 -j REDIRECT --to-port 8888
sudo iptables -I INPUT 5 -p tcp --dport 8888 -j ACCEPT
sudo bash -c "iptables-save > /etc/sysconfig/iptables"

如果开启了 SELinux ，查看允许的 http 端口

sudo semanage port -l | grep http_port_t

输出结果

http_port_t                    tcp      80, 81, 443, 488, 8008, 8009, 8443, 9000

8888 端口不在允许范围内，添加 8888 端口

sudo semanage port -a -t http_port_t -p tcp 8888

配置完成之后，可以修改 Nginx 配置文件的监听端口为 8888 ，然后直接访问服务器的 80 端口即可

# 配置文件语法高亮

# 针对所有用户

sudo wget http://www.vim.org/scripts/download_script.php?src_id=19394 -O /usr/share/vim/vim74/syntax/nginx.vim
sudo vi /usr/share/vim/vim74/filetype.vim

在 filetype.vim 追加一行，添加 nginx 配置文件路径

au BufRead,BufNewFile /etc/nginx/*,/etc/nginx/conf.d/*,/home/nginx/u01/nginx-1.13.12/conf/*,/home/nginx/u01/nginx-1.13.12/conf.d/* if &ft == '' | setfiletype nginx | endif

TIP

nginx 配置文件路径一定是绝对路径，不能包含软链接

# 针对普通用户

mkdir -p ~/.vim/syntax/
wget http://www.vim.org/scripts/download_script.php?src_id=19394 -O ~/.vim/syntax/nginx.vim
vi ~/.vim/filetype.vim

同样添加 nginx 配置文件路径

au BufRead,BufNewFile /etc/nginx/*,/etc/nginx/conf.d/*,/home/www/u01/nginx-1.13.12/conf/*,/home/www/u01/nginx-1.13.12/conf.d/* if &ft == '' | setfiletype nginx | endif

# 给主进程发送信号

快速关闭 Nginx

nginx -s stop

优雅关闭 Nginx ，停止接收连接请求，并等待已接收连接处理完成之后才退出

nginx -s quit

Nginx 服务运行时重新打开日志，一般在手动切割日志时使用

nginx -s reopen

重载 Nginx 配置

nginx -s reload

# 启动时指定配置文件

nginx -c /usr/local/nginx/conf/nginx.conf
