# 安装

# 依赖

sudo yum install -y prce pcre-devel openssl openssl-devel policycoreutils-python gcc gcc-c++ make

# 运行用户

创建一个普通用户来运行 Nginx ，而不使用 root

useradd nginx
passwd nginx

设置 nginx 用户的最大打开文件数和最大进程数，设置完成后最好重启一遍系统

sudo vi /etc/security/limits.d/20-nproc.conf
nginx   soft    nproc   65535
nginx   hard    nproc   65535
nginx   soft    nofile  65535
nginx   hard    nofile  65535

# 编译

nginx_path=/home/nginx/u01/nginx
mkdir -p $nginx_path/src
cd $nginx_path/src
wget -4 http://nginx.org/download/nginx-1.18.0.tar.gz
tar xf nginx-1.18.0.tar.gz
cd nginx-1.18.0
./configure \
--prefix=$nginx_path \
--user=nginx \
--group=nginx \
--pid-path=$nginx_path/run/nginx.pid \
--lock-path=$nginx_path/run/nginx.lock \
--error-log-path=$nginx_path/logs/error.log \
--http-log-path=$nginx_path/logs/access.log \
--with-http_ssl_module \
--with-http_stub_status_module \
--with-http_gzip_static_module \
--with-http_stub_status_module \
--with-pcre \
--with-file-aio \
--with-http_realip_module \
--with-http_v2_module \
--with-stream

make && make install

# 服务端口

非 root 用户不能使用 1024 以下的端口，有2种解决办法

# sudo

sudo `which nginx`

# iptables端口转发

重定向 80 端口到 8888 端口

sudo iptables -t nat -A PREROUTING -i eth0 -p tcp --dport 80 -j REDIRECT --to-port 8888
sudo iptables -I INPUT 5 -p tcp --dport 8888 -j ACCEPT
sudo bash -c "iptables-save > /etc/sysconfig/iptables"

如果开启了 SELinux ，查看允许的 http 端口

sudo semanage port -l | grep http_port_t

输出结果

http_port_t                    tcp      80, 81, 443, 488, 8008, 8009, 8443, 9000

8888 端口不在允许范围内，添加 8888 端口

sudo semanage port -a -t http_port_t -p tcp 8888

配置完成之后，可以修改 Nginx 配置文件的监听端口为 8888 ，然后直接访问服务器的 80 端口即可

# 配置文件语法高亮

# 针对所有用户

sudo wget http://www.vim.org/scripts/download_script.php?src_id=19394 -O /usr/share/vim/vim74/syntax/nginx.vim
sudo vi /usr/share/vim/vim74/filetype.vim

在 filetype.vim 追加一行，添加 nginx 配置文件路径

au BufRead,BufNewFile /etc/nginx/*,/etc/nginx/conf.d/*,/home/nginx/u01/nginx-1.13.12/conf/*,/home/nginx/u01/nginx-1.13.12/conf.d/* if &ft == '' | setfiletype nginx | endif

TIP

nginx 配置文件路径一定是绝对路径，不能包含软链接

# 使用日志分割功能

安装cronolog日志分割工具，需要配置epel源

yum -y install cronolog

日志分割配置

http {
    ...
    log_format wwwlogs '$host,,$remote_addr,,SERVER_IPADDR,,$request_time_msec,,$remote_user,,$time_iso8601,,$request_method,,$scheme://$host:$server_port$request_uri,,$status,,$body_bytes_sent,,$http_referer,,$http_user_agent,,$http_x_forwarded_for,,PROJECT_CODE';
    ...
    
    server {
        access_log  "pipe:/usr/sbin/cronolog /5w/nginx/logs/ywtg/%Y-%m-%d-access_http_log" wwwlogs;
        error_log   "pipe:/usr/sbin/cronolog /5w/nginx/logs/ywtg/%Y-%m-%d-error_log" warn;
    }
}

TIP

注意：日志路径必须是绝对路径，相对路径会生成日志到Nginx可执行文件目录下

# 给主进程发送信号

快速关闭 Nginx

nginx -s stop

优雅关闭 Nginx ，停止接收连接请求，并等待已接收连接处理完成之后才退出

nginx -s quit

Nginx 服务运行时重新打开日志，一般在手动切割日志时使用

nginx -s reopen

重载 Nginx 配置

nginx -s reload

# 启动时指定配置文件

nginx -c /usr/local/nginx/conf/nginx.conf

# rtmp配置

rtmp.conf

rtmp_auto_push on;
rtmp_auto_push_reconnect 1s;
rtmp_socket_dir /tmp;

rtmp {
    out_queue           4096;
    out_cork            8;
    timeout             15s;
    drop_idle_publisher 15s;

    log_interval 5s; #log 模块在 libfdk_aacess.log 中记录日志的间隔时间，对调试非常有用
    log_size     1m; #log 模块用来记录日志的缓冲区大小

    server {
        listen 1935;
#        server_name www.test.*; #用于虚拟主机名后缀通配

        application jdp {
            live on;
            gop_cache on; #打开 GOP 缓存，减少首屏等待时间
            #record all;
            #record_unique on;
            #record_max_size 500M;
            #record_path /5w/nginx/html/rtmp/record;
            #record_suffix -%Y-%m-%d-%H_%M_%S.flv;

            exec /usr/local/ffmpeg/bin/ffmpeg -i rtmp://localhost:1935/jdp/$name -i /5w/nginx/html/rtmp/images/td1.png -filter_complex "overlay=10:10,split=1[td1]" -map '[td1]' -map 0:a -s 1280x720 -c:v libx264 -c:a aac -g 30 -r 30 -tune zerolatency -preset veryfast -crf 23 -f flv rtmp://localhost:1935/live/$name_td1; 

            exec /usr/local/ffmpeg/bin/ffmpeg -i rtmp://localhost:1935/jdp/$name -i /5w/nginx/html/rtmp/images/td2.png -filter_complex "overlay=10:10,split=1[td2]" -map '[td2]' -map 0:a -s 1280x720 -c:v libx264 -c:a aac -g 30 -r 30 -tune zerolatency -preset veryfast -crf 23 -f flv rtmp://localhost:1935/live/$name_td2;

            exec /usr/local/ffmpeg/bin/ffmpeg -i rtmp://localhost:1935/jdp/$name -i /5w/nginx/html/rtmp/images/td3.png -filter_complex "overlay=10:10,split=1[td3]" -map '[td3]' -map 0:a -s 1280x720 -c:v libx264 -c:a aac -g 30 -r 30 -tune zerolatency -preset veryfast -crf 23 -f flv rtmp://localhost:1935/live/$name_td3;
        }

        application live {
            live on;
            gop_cache on;
}

        application hls {
            live on;
            hls on;
            hls_path /5w/nginx/html/rtmp/hls;
        }

        application dash {
            live on;
            dash on;
            dash_path /5w/nginx/html/rtmp/dash;
        }
    }

}
