# 使用PHP-FPM

安装 php ，配置 Nginx + PHP-FPM 运行环境

# 依赖

sudo yum install -y epel-release.noarch
sudo yum install -y libcurl-devel bzip2-devel libmcrypt-devel openssl-devel libxml2-devel libjpeg-turbo-devel libpng-devel libvpx-devel freetype-devel libicu-devel m4 autoconf

# 运行用户

useradd www
passwd www

设置 www 用户的最大打开文件数和最大进程数，设置完成后最好重启一遍系统

vi /etc/security/limits.d/20-nproc.conf
www   soft    nproc   65535
www   hard    nproc   65535
www   soft    nofile  65535
www   hard    nofile  65535

# 编译

mkdir -p /home/www/u01/php-5.6.36/{src,etc}
cd /home/www/u01/php-5.6.36/src
wget http://hk2.php.net/distributions/php-5.6.36.tar.gz
tar xf php-5.6.36.tar.gz
cd php-5.6.36
./configure \
--prefix=/home/www/u01/php-5.6.36 \
--with-openssl \
--enable-mbstring \
--enable-sockets \
--with-freetype-dir \
--with-jpeg-dir \
--with-gd \
--with-png-dir \
--enable-xml \
--enable-exif \
--with-zlib \
--with-mcrypt \
--with-bz2 \
--with-mhash \
--with-config-file-path=/home/www/u01/php-5.6.36/etc \
--with-config-file-scan-dir=/home/www/u01/php-5.6.36/etc/php.d \
--with-mysql=mysqlnd \
--with-mysqli=mysqlnd \
--with-pdo-mysql=mysqlnd \
--enable-fpm \
--with-curl \
--enable-intl \
--enable-zip \
--with-fpm-user=www \
--with-fpm-group=www

make && make install

# 软链接

为 PHP 创建一条软链接，方便管理

ln -s /home/www/u01/php-5.6.36 /home/www/u01/php

拷贝 PHP 配置文件和启动程序

pwd
/home/www/u01/php/src/php-5.6.36

cp php.ini-production /home/www/u01/php/etc/php.ini
cp sapi/fpm/init.d.php-fpm /home/www/u01/php/sbin/php-fpmd
chmod u+x /home/www/u01/php/sbin/php-fpmd
sudo ln -s /home/www/u01/php/sbin/php-fpm /usr/bin/php-fpm
sudo ln -s /home/www/u01/php/sbin/php-fpmd /usr/bin/php-fpmd
sudo bash -c 'echo "export PATH=/home/www/u01/php/bin:ATH" > /etc/profile.d/php.sh'

# 调整php-fpm.conf参数

pwd
/home/www/u01/php

cp etc/php-fpm.conf.default etc/php-fpm.conf
vi etc/php-fpm.conf
pm.max_children = 50
pm.start_servers = 5
pm.min_spare_servers = 2
pm.max_spare_servers = 8

# 启动php-fpm

php-fpmd start

[09-May-2018 12:43:20] NOTICE: [pool www] 'group' directive is ignored when FPM i
 done

# 安装PHP扩展

安装 GD 扩展并添加 jpeg 支持示例：

安装依赖

sudo yum install libjpeg-devel libpng-devel

编译

pwd
/home/www/u01/php/src/php-5.6.36/ext/gd

phpize
Configuring for:
PHP Api Version:         20131106
Zend Module Api No:      20131226
Zend Extension Api No:   220131226
./configure --with-php-config=/home/www/u01/php/bin/php-config --with-jpeg-dir=/usr/lib64
make && make install

TIP

如果之前编译失败，可在 make && make install 之前运行 make clean ，然后重新编译

启用扩展

在 php.ini 中启用扩展

extension=gd.so

重启PHP-FPM

php-fpmd restart

使用非yum安装的jpeg

如果需要使用更高版本的 jepg ，可单独编译安装后，在安装 PHP 扩展时指定该路径

mkdir -p ~/u01/jpeg-9c/src
cd ~/u01/jpeg-9c/src
wget http://www.ijg.org/files/jpegsrc.v9c.tar.gz
tar xf jpegsrc.v9c.tar.gz
cd jpeg-9c
./configure --prefix=/home/www/u01/jpeg-9c --enable-shared --enable-static
make && make install

使用单独安装的 jepg 路径

./configure --with-php-config=/home/www/u01/php/bin/php-config --with-jpeg-dir=/home/www/u01/jpeg-9c

# Nginx配置

upstream php {
    server 127.0.0.1:9000;
}

server {
    listen 8888;
    server_name www.example.com;
    access_log  logs/access.log  main;
    root html;
    index index.php index.html;
    location ~ \.php$ {
        include fastcgi.conf;
        fastcgi_pass php;
        #fastcgi_pass 127.0.0.1:9000;
    }
}
